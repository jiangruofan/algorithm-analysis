```
def minimum_cost(s, cost):
    f = [[float("Inf")]*len(s) for k in range(1<<len(s))]
    
    for k in range(1<<len(s)):
        for i in range(len(s)):
            if (k & (1 << i)) != 0:
                if (k & ~(1 << i)) == 0:
                    f[k][i] = 0
                else:
                    for j in range(len(s)):
                        if j != i:
                            f[k][i] = min(f[k][i], f[k & ~(1 << i)][j] + cost[j][i])
    
    return min(f[(1<<len(s))-1])
```
